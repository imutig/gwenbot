        // Mock Data
        const data = {
            session: [
                { name: "Chaton_Râleur", points: 1000, rank: 1 },

                { name: "PixelPrincess", points: 950, rank: 3 },

                { name: "DarkSlayer", points: 610, rank: 6 },
                { name: "BulleDeGomme", points: 400, rank: 7 },
                { name: "TeaTime", points: 350, rank: 8 },
            ],
            global: [
                { name: "GodOfWords", points: 25400, rank: 1 },
                { name: "CemantixMaster", points: 22100, rank: 2 },
                { name: "Chaton_Râleur", points: 18900, rank: 3 },
                { name: "NoLifePlayer", points: 15400, rank: 4 },
                { name: "MysterySolver", points: 12300, rank: 5 },
            ]
        };

        function renderLeaderboard(type) {
            const container = document.getElementById('leaderboardContainer');
            container.innerHTML = '';
            
            data[type].forEach((player, index) => {
                const item = document.createElement('div');
                item.className = 'leaderboard-item flex items-center justify-between p-4 bg-white/20 backdrop-blur-sm shadow-sm';
                item.style.animationDelay = `${index * 0.1}s`;
                
                let rankClass = player.rank <= 3 ? `rank-${player.rank}` : 'bg-pink-100 text-pink-500';
                let icon = '';
                if(player.rank === 1) icon = '<i data-lucide="crown" class="w-3 h-3 absolute -top-2 -right-1 text-yellow-500 rotate-12"></i>';

                item.innerHTML = `
                    <div class="flex items-center gap-4">
                        <div class="relative">
                            <div class="rank-badge ${rankClass}">
                                ${player.rank}
                            </div>
                            ${icon}
                        </div>
                        <span class="font-bold text-[#8B4558]">${player.name}</span>
                    </div>
                    <div class="flex items-center gap-2">
                        <span class="font-bold text-pink-600">${player.points.toLocaleString()}</span>
                        <span class="text-[10px] text-pink-400 font-bold uppercase">pts</span>
                    </div>
                `;
                container.appendChild(item);
            });
            
            lucide.createIcons();
        }

        function switchTab(type) {
            const sessionBtn = document.getElementById('sessionBtn');
            const globalBtn = document.getElementById('globalBtn');

            if (type === 'session') {
                sessionBtn.classList.add('active');
                sessionBtn.classList.remove('text-pink-400');
                globalBtn.classList.remove('active');
                globalBtn.classList.add('text-pink-400');
            } else {
                globalBtn.classList.add('active');
                globalBtn.classList.remove('text-pink-400');
                sessionBtn.classList.remove('active');
                sessionBtn.classList.add('text-pink-400');
            }

            renderLeaderboard(type);
        }

        // Initialize
        window.onload = () => {
            renderLeaderboard('session');
            lucide.createIcons();
        }
    </script>
